## Zustand Store模板
```typescript
interface MyStore {
  // 状态
  items: Item[];

  // 操作
  addItem: (item: Item) => void;
  removeItem: (id: string) => void;
}

const useMyStore = create<MyStore>((set) => ({
  items: [],

  addItem: (item) =>
    set((state) => ({
      items: [...state.items, item]
    })),

  removeItem: (id) =>
    set((state) => ({
      items: state.items.filter((item) => item.id !== id)
    }))
}));
```

## Java通信封装
```typescript
class JavaBridge {
  async invoke(action: string, params?: any): Promise<any> {
    const queryId = ++this.id;

    return new Promise((resolve, reject) => {
      // 设置超时30秒
      const timeout = setTimeout(() => {
        reject(new Error(`Timeout: ${action}`));
      }, 30000);

      // 存储pending请求
      this.pending.set(queryId, { resolve, reject });

      // 调用Java
      window.ccQuery?.invoke(JSON.stringify({
        queryId,
        action,
        params
      }));
    });
  }

  // 具体API
  async sendMessage(content: string) {
    return this.invoke('sendMessage', { content });
  }

  // 事件监听
  on(event: string, handler: Function) {
    window.ccEvents?.on(event, handler);
  }
}
```
